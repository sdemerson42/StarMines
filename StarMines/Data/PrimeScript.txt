setPos 400 300;

def switch;
rand vx (0-5) 5;
playAnim "right";
ifLess vx 0 ^left;
rand vy (0-5) 5;
setDir vx vy;
setSpeed 60;
let counter 0;
rand limit 30 70;
let slow (limit/2);

def loop;
posBound 0 770 0 570;
inc counter 1;
pause;
ifEquals counter limit >switch;
ifEquals counter slow ^brakes;
jump >loop;

def brakes;
setSpeed 20;
retSub;

def left;
playAnim "left";
retSub;

def collision;
sendToCaller "encroach";
jump >switch;
