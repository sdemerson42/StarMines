setSpeed 20;
rand x 0 770;
rand y 0 570;
setPos x y;
playAnim "active";

def find;
let counter 0;
setTargetTag "RedGuy" "first";
letTargetPos tx ty;
letPos x y;
setDir (tx-x) (ty-y);

def loop;
inc counter 1;
posBound 0 770 0 570;
pause;
if counter > 20 @find;
jump @loop;

def collision;
ifCallerTag "RedGuy" @die;
jump @loop;

def die;
despawn;
halt;
